# 梦中日记本 实施计划

> **目标：** 搭建一个白底墨韵的语音 AI 日记 PWA 应用，用户上传照片→粒子点彩画→AI 对话→生成日记→滑动浏览

> **架构：** 纯前端单页应用，20 个 JS/CSS 模块文件。用浏览器自带语音识别（免费）、Gemini API 润色文字、Canvas 2D 画粒子、IndexedDB 存数据

> **技术：** HTML + CSS + JS (ES Modules)、Gemini API、Web Speech API、Canvas 2D、IndexedDB

```javascript
    timer = setTimeout(() => fn(...args), ms);
  };
```

```javascript
// js/db.js
// 用 IndexedDB 存日记。把它想象成一个本地小数据库，比 localStorage 能存更多东西

const DB_NAME = 'DreamDiaryDB';
const DB_VERSION = 1;

let db = null;

// 打开数据库（像打开一个文件夹，如果不存在就创建）
export function openDB() {
  return new Promise((resolve, reject) => {
    if (db) return resolve(db); // 已经打开过了，直接用
    
    const request = indexedDB.open(DB_NAME, DB_VERSION);
    
    // 数据库第一次创建或版本升级时，建表
    request.onupgradeneeded = (e) => {
      const database = e.target.result;
      if (!database.objectStoreNames.contains('diaries')) {
        const store = database.createObjectStore('diaries', { keyPath: 'id' });
        store.createIndex('createdAt', 'createdAt', { unique: false });
        store.createIndex('tags', 'tags', { unique: false, multiEntry: true });
        store.createIndex('mood', 'mood', { unique: false });
      }
    };
    
    request.onsuccess = (e) => {
      db = e.target.result;
      resolve(db);
    };
    
    request.onerror = () => reject(new Error('无法打开数据库'));
  });
}

// 保存一篇日记（新增或更新）
export async function saveDiary(diary) {
  const database = await openDB();
  return new Promise((resolve, reject) => {
    const tx = database.transaction('diaries', 'readwrite');
    const store = tx.objectStore('diaries');
    store.put(diary); // put = 有就更新，没有就新增
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(new Error('保存失败'));
  });
}

// 获取一篇日记
export async function getDiary(id) {
  const database = await openDB();
  return new Promise((resolve, reject) => {
    const tx = database.transaction('diaries', 'readonly');
    const store = tx.objectStore('diaries');
    const request = store.get(id);
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(new Error('读取失败'));
  });
}

// 获取所有日记（按时间倒序，最新的排前面）
export async function getAllDiaries() {
  const database = await openDB();
  return new Promise((resolve, reject) => {
    const tx = database.transaction('diaries', 'readonly');
    const store = tx.objectStore('diaries');
    const index = store.index('createdAt');
    const request = index.openCursor(null, 'prev'); // prev = 倒序
    const results = [];
    request.onsuccess = (e) => {
      const cursor = e.target.result;
      if (cursor) {
        results.push(cursor.value);
        cursor.continue();
      } else {
        resolve(results);
      }
    };
    request.onerror = () => reject(new Error('读取列表失败'));
  });
}

// 删除一篇日记
export async function deleteDiary(id) {
  const database = await openDB();
  return new Promise((resolve, reject) => {
    const tx = database.transaction('diaries', 'readwrite');
    const store = tx.objectStore('diaries');
    store.delete(id);
    tx.oncomplete = () => resolve();
    tx.onerror = () => reject(new Error('删除失败'));
  });
}

// 按标签筛选日记
export async function getDiariesByTag(tag) {
  const database = await openDB();
  return new Promise((resolve, reject) => {
    const tx = database.transaction('diaries', 'readonly');
    const store = tx.objectStore('diaries');
    const index = store.index('tags');
    const request = index.getAll(tag);
    request.onsuccess = () => resolve(request.result);
    request.onerror = () => reject(new Error('筛选失败'));
  });
}
```

```javascript
const API_BASE = 'https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash';

// AI 的"人设"——告诉它该怎么写日记
const DIARY_SYSTEM_PROMPT = `你是一位温暖、细腻的文学编辑。用户会给你一段口语化的语音转写文本，
你需要将其转化为一篇精美的日记。

请按以下 JSON 格式返回（只返回 JSON，不要其他内容）：

{
  "title": "诗意的标题（10字以内）",
  "content": "润色后的日记正文，温暖文学的语调，保留原意的同时让文字更优美",
  "tags": ["情绪标签1", "情绪标签2", "情绪标签3"],
  "mood": "整体情绪（开心/平静/忧伤/焦虑/期待/感动/思念/迷茫）",
  "quote": "从日记中提取一句最有感触的话"
}

要求：
1. 保持第一人称，不要改变用户的原意
2. 风格温暖治愈，像村上春树或吉本芭娜娜的散文语调
3. 适当分段，让阅读节奏舒适
4. 标签要准确反映情绪，不要泛化`;

// 获取 API Key（优先从设置中读，没有就用占位符）
export function getApiKey() {
  try {
    const settings = JSON.parse(localStorage.getItem('dream-diary-settings') || '{}');
    return settings.geminiApiKey || '';
  } catch {
    return '';
  }
}

// 把用户的口语文字发给 Gemini，拿回润色后的日记 JSON
export async function polishDiary(rawText) {
  const apiKey = getApiKey();
  if (!apiKey) {
    throw new Error('请先在设置中填入 Gemini API Key');
  }

  const response = await fetch(`${API_BASE}:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      system_instruction: { parts: [{ text: DIARY_SYSTEM_PROMPT }] },
      contents: [{ parts: [{ text: rawText }] }]
    })
  });

  if (!response.ok) {
    const err = await response.json().catch(() => ({}));
    throw new Error(err.error?.message || `API 调用失败（${response.status}）`);
  }

  const data = await response.json();
  const text = data.candidates?.[0]?.content?.parts?.[0]?.text;
  if (!text) throw new Error('AI 返回为空，请重试');

  // 清理可能包裹的 markdown 代码块标记
  const cleanJson = text.replace(/```json\n?|```/g, '').trim();
  return JSON.parse(cleanJson);
}

// 聊天对话：就某篇日记和 AI 聊天
const CHAT_SYSTEM_PROMPT = `你是一位温柔贴心的朋友，正在和用户聊ta刚写的日记。
请用温暖、自然的语气回应，不要像机器人。可以适当提问、共情或分享感受。

  const apiKey = getApiKey();
  if (!apiKey) throw new Error('请先设置 API Key');

  // 构建对话上下文
  const messages = [
    { role: 'user', parts: [{ text: `以下是我刚写的日记：\n\n${diaryContent}\n\n基于这篇日记，和我聊聊吧。` }] },
    { role: 'model', parts: [{ text: '好的，我读了你的日记。有什么想聊的吗？' }] },

    { role: 'user', parts: [{ text: userMessage }] }
```

### 任务 4：实现粒子引擎（照片 → 墨点画）

**文件：** 新建 `js/particles.js`

这是整个应用视觉的灵魂。简单说：把一张照片"拆"成几千个小墨点，每个墨点根据照片对应位置的颜色决定自己的大小和浓淡。

- [ ] **Step 1：写粒子引擎核心**

```javascript
// js/particles.js
// 粒子引擎：把照片变成几千个会微微飘动的墨点

// 所有粒子的数据
let particles = [];
let canvas = null;
let ctx = null;
let animationId = null;
let currentImageData = null; // 当前照片的像素数据

// ===== 第一部分：照片分析 =====

// 从图片 URL 加载并分析像素
export function loadImageForParticles(imageUrl) {
  return new Promise((resolve, reject) => {
    const img = new Image();
    img.crossOrigin = 'anonymous';
    img.onload = () => {
      // 把图片画到临时画布上，读取每个像素的颜色
      const tempCanvas = document.createElement('canvas');
      const size = 200; // 统一缩放到 200x200 分析
      tempCanvas.width = size;
      tempCanvas.height = size;
      const tempCtx = tempCanvas.getContext('2d');
      
      // 按比例缩放绘制
      const scale = Math.min(size / img.width, size / img.height);
      const w = img.width * scale;
      const h = img.height * scale;
      const x = (size - w) / 2;
      const y = (size - h) / 2;
      tempCtx.drawImage(img, x, y, w, h);
      
      currentImageData = tempCtx.getImageData(0, 0, size, size);
      resolve(currentImageData);
    };
    img.onerror = () => reject(new Error('图片加载失败'));
    img.src = imageUrl;
  });
}

// 从像素数据生成粒子
export function generateParticles(imageData, particleCount = 2000) {
  const { data, width, height } = imageData;
  particles = [];
  
  // 在图片区域内随机撒点，每个点读取对应位置的颜色
  for (let i = 0; i < particleCount; i++) {
    const px = Math.floor(Math.random() * width);
    const py = Math.floor(Math.random() * height);
    const pixelIndex = (py * width + px) * 4;
    
    const r = data[pixelIndex];
    const g = data[pixelIndex + 1];
    const b = data[pixelIndex + 2];
    
    // 转成灰度（0=纯黑, 255=纯白）
    const gray = 0.299 * r + 0.587 * g + 0.114 * b;
    
    // 白底版本：深色区域 = 浓墨大粒子，浅色区域 = 淡墨小粒子
    particles.push({
      x: px,
      y: py,
      originX: px,      // 原始位置（粒子飘动后要回到这个位置附近）
      originY: py,
      gray: gray,
      size: 2 + (1 - gray / 255) * 4,   // 深色大（最大6px），浅色小（最小2px）
      opacity: 0.3 + (1 - gray / 255) * 0.6, // 深色浓（最高0.9），浅色淡（最低0.3）
      phase: Math.random() * Math.PI * 2, // 随机初始相位（让每个粒子飘动节奏不同）
      speed: 0.3 + Math.random() * 0.7,   // 飘动速度随机
    });
  }
  
  return particles;
}

// ===== 第二部分：动画渲染 =====

// 简化版 Perlin Noise（让粒子像水草一样自然摇摆，而不是机械振动）
// 这里用一个简化的正弦波叠加来模拟
function noiseOffset(x, y, time, phase, speed) {
  const t = time * 0.001; // 时间单位换成秒
  return {
    dx: Math.sin(x * 0.05 + t * speed + phase) * 2 + Math.cos(y * 0.03 + t * 0.7 * speed) * 1.5,
    dy: Math.cos(y * 0.05 + t * speed + phase) * 2 + Math.sin(x * 0.03 + t * 0.7 * speed) * 1.5,
  };
}

// 初始化 Canvas
export function initParticleCanvas(containerEl, width, height) {
  canvas = document.createElement('canvas');
  canvas.width = width;
  canvas.height = height;
  canvas.style.width = '100%';
  canvas.style.height = '100%';
  canvas.style.borderRadius = '18px';
  containerEl.appendChild(canvas);
  ctx = canvas.getContext('2d');
}

// 开始动画循环
export function startAnimation(audioVolume = 0) {
  if (animationId) return;
  
  let startTime = performance.now();
  
  function loop(now) {
    const elapsed = now - startTime;
    renderFrame(elapsed, audioVolume);
    animationId = requestAnimationFrame(loop);
  }
  
  animationId = requestAnimationFrame(loop);
}

// 渲染一帧
function renderFrame(time, audioVolume) {
  if (!ctx || !canvas) return;
  
  // 清空画布
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  
  // 音量影响幅度（安静时几乎不动，音乐响时多飘一点）
  const volumeBoost = 1 + audioVolume * 2;
  
  for (const p of particles) {
    // 计算当前偏移
    const noise = noiseOffset(p.originX, p.originY, time, p.phase, p.speed);
    const x = p.originX + noise.dx * volumeBoost;
    const y = p.originY + noise.dy * volumeBoost;
    
    // 画粒子（径向渐变圆，中心实→边缘虚）
    ctx.beginPath();
    ctx.arc(x, y, p.size / 2, 0, Math.PI * 2);
    
    // 用径向渐变让边缘变软
    const gradient = ctx.createRadialGradient(x, y, 0, x, y, p.size / 2);
    gradient.addColorStop(0, `rgba(0,0,0,${p.opacity})`);
    gradient.addColorStop(1, 'rgba(0,0,0,0)');
    ctx.fillStyle = gradient;
    ctx.fill();
  }
}

// 停止动画
export function stopAnimation() {
  if (animationId) {
    cancelAnimationFrame(animationId);
    animationId = null;
  }
}

// 清理
export function destroyParticles() {
  stopAnimation();
  particles = [];
  if (canvas) {
    canvas.remove();
    canvas = null;
    ctx = null;
  }
}

// 获取当前粒子数量
export function getParticleCount() {
  return particles.length;
}
```

```javascript
// js/recorder.js
// 用浏览器自带的语音识别，把说的话实时转成文字
// 不需要联网，不需要 API Key，浏览器自带功能

let recognition = null;
let isRecording = false;
let currentTranscript = '';
let onResultCallback = null;
let onStateChangeCallback = null;

// 检查浏览器是否支持语音识别
export function isSpeechSupported() {
  return 'SpeechRecognition' in window || 'webkitSpeechRecognition' in window;
}

// 开始录音
export function startRecording(onResult, onStateChange) {
  if (!isSpeechSupported()) {
    throw new Error('您的浏览器不支持语音识别，请使用 Chrome 浏览器');
  }
  
  const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
  recognition = new SpeechRecognition();
  recognition.lang = 'zh-CN';          // 中文识别
  recognition.interimResults = true;   // 实时显示中间结果
  recognition.continuous = true;       // 持续监听，不自动停止
  recognition.maxAlternatives = 1;
  
  currentTranscript = '';
  onResultCallback = onResult;
  onStateChangeCallback = onStateChange;
  
  // 识别到语音时
  recognition.onresult = (event) => {
    let interim = '';
    let final = '';
    
    for (let i = event.resultIndex; i < event.results.length; i++) {
      const text = event.results[i][0].transcript;
      if (event.results[i].isFinal) {
        final += text;
      } else {
        interim += text;
      }
    }
    
    if (final) currentTranscript += final;
    // 回调：把当前所有已识别的文字传出去
    onResultCallback(currentTranscript + interim, !!interim);
  };
  
  // 识别出错时
  recognition.onerror = (event) => {
    if (event.error === 'no-speech') return; // 没人说话不算错
    if (event.error === 'aborted') return;   // 主动停止不算错
    onStateChangeCallback?.('error', event.error);
  };
  
  // 开始/结束时的状态通知
  recognition.onstart = () => {
    isRecording = true;
    onStateChangeCallback?.('start');
  };
  
  recognition.onend = () => {
    isRecording = false;
    onStateChangeCallback?.('stop', currentTranscript);
  };
  
  recognition.start();
}

// 停止录音
export function stopRecording() {
  if (recognition && isRecording) {
    recognition.stop();
  }
}

// 获取当前累积的文字
export function getTranscript() {
  return currentTranscript;
}

// 是否正在录音
export function isCurrentlyRecording() {
  return isRecording;
}

// 重置（开始新录音前清空之前的文字）
export function resetTranscript() {
  currentTranscript = '';
}
```

```javascript
  stop();
  
  utterance = new SpeechSynthesisUtterance(text);
  utterance.lang = 'zh-CN';

  utterance.onstart = () => { isSpeaking = true; };
  utterance.onend = () => { isSpeaking = false; };
  utterance.onerror = () => { isSpeaking = false; };

}
```

**文件：** 新建 `manifest.json`、`sw.js`

- [ ] **Step 1：写 PWA 配置**

```json
// manifest.json
{
  "name": "梦中日记本",
  "short_name": "日记本",
  "description": "AI 语音交互日记 — 用声音记录生活",
  "start_url": "/",
  "display": "standalone",
  "background_color": "#FFFFFF",
  "theme_color": "#FFFFFF",
  "icons": []
}
```

```javascript
// sw.js
// Service Worker：缓存静态文件，离线也能打开
const CACHE_NAME = 'dream-diary-v1';

const FILES_TO_CACHE = [
  '/',
  '/index.html',
  '/css/main.css',
  '/css/components.css',
  '/js/app.js',
  '/js/utils.js',
  '/js/db.js',
  '/js/gemini.js',
  '/js/recorder.js',
  '/js/tts.js',
  '/js/diary.js',
  '/js/ui.js',
  '/js/particles.js',
  '/js/settings.js',
];

// 安装时缓存
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(FILES_TO_CACHE))
  );
});

// 请求时优先用缓存
self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => response || fetch(event.request))
  );
});
```
